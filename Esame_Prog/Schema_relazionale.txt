Vincoli esterni: per ogni istanza 'r' di Ricovero:
r.terminato = 'True' se e solo se r.fine è valorizzato.
r.fine > r.inizio.

		Schema relazionale delle base dati
	
Paziente(__cf:varchar__, nome:varchar, cognome:varchar)

Ricovero(__id:Serial__, inizio:TimeStamp,terminato:Boolean,fine:TimeStamp*,paziente:varchar)
foreign key: (paziente) references Paziente(cf) 
CHECK (WHEN terminato is False THEN fine is null,
	ELSE fine is not null AND fine > inizio)
END

foreign key: (medico) references Medico(codice) 
foreign key: (ricovero) references Ricovero(id)
